document.addEventListener("DOMContentLoaded", loadDashboard); async function loadDashboard() { const r = await fetch("/api/events"); const games = await r.json(); const table = document.getElementById("dashboard"); table.innerHTML = `
Matchup EV Away EV Home Top EV Kickoff
`; const body = table.querySelector("tbody"); games.sort((a, b) => b.bestEV - a.bestEV); for (const g of games) { const row = document.createElement("tr"); row.innerHTML = ` ${g.away_team} @ ${g.home_team} ${(g.ev.awayML * 100).toFixed(1)}% ${(g.ev.homeML * 100).toFixed(1)}% ${(g.bestEV * 100).toFixed(1)}% ${new Date(g.commence_time).toLocaleString()} `; body.appendChild(row); } }
